Chapter 6 Tagging Table:

dbExecute(nff, "CREATE TABLE tagging (
tag_id INTEGER PRIMARY KEY,
lamb_id varchar(5),
serial_number TEXT,
freq VARCHAR(6),
rancher TEXT,
tag_date TEXT,
sex TEXT,
breed TEXT,
FOREIGN KEY(lamb_id) REFERENCES master(lamb_id)
FOREIGN KEY(serial_number) REFERENCES collars(serial_number)
);")
tagging <- read.csv("~/WILD6900sp21/nf_final_project/tagging.csv") 
dbWriteTable(nff, "tagging", tagging, append = T)
